\begin{frame}{Model Setup}
\phantomsection\label{model-setup}
\BerkeleyLightMode

\begin{assumptionbox}{Value Function}
$$V(s_t) = \max_{d_t \in \{0, 1, 2\}} \left[ \pi(s_t, d_t) + \beta \mathbb{E}[V(s_{t+1}) \mid s_t, d_t] \right]$$
\end{assumptionbox}

\textbf{Choice Set (\(d_t\)):}

\begin{itemize}
\tightlist
\item
  \(d_t = 0\): \textbf{Continue} --- status quo utility, stochastic age
  transition
\item
  \(d_t = 1\): \textbf{Retrofit} --- incur fixed cost \(K_R\), reset
  regulatory state
\item
  \(d_t = 2\): \textbf{Exit} --- claim terminal scrap value \(K_E\),
  permanent absorption
\end{itemize}
\end{frame}

\begin{frame}{4-Parameter Baseline: State Space \& Bellman Equation}
\phantomsection\label{parameter-baseline-state-space-bellman-equation}
\BerkeleyLightMode

\begin{assumptionbox}{Dynamic Programming Framework — Homogeneous Costs}
$$V(s_t) = \max_{d_t \in \{0, 1, 2\}} \left[ u(s_t, d_t) + \beta \int V(s_{t+1}) dP(s_{t+1} \mid s_t, d_t) \right]$$
\end{assumptionbox}

\textbf{State Space:} \(s_t = (a_t, \epsilon_t)\) --- tank age and
i.i.d. Type I EV profitability shock.

Identical compliance cost structures assumed across all facilities.
\end{frame}

\begin{frame}{4-Parameter Baseline: Utility Specification}
\phantomsection\label{parameter-baseline-utility-specification}
\BerkeleyLightMode

\(\theta = (\alpha, \gamma, K_R, K_E)\)

\begin{itemize}
\tightlist
\item
  \textbf{Continue (\(d_t = 0\)):}
  \(u = \alpha - \gamma a_t + \epsilon_{0t}\) \(\quad\) \emph{(base
  profit, age penalty)}
\item
  \textbf{Retrofit (\(d_t = 1\)):} \(u = \alpha - K_R + \epsilon_{1t}\)
  \(\quad\) \emph{(fixed upgrade cost, resets \(a_{t+1}=0\))}
\item
  \textbf{Exit (\(d_t = 2\)):} \(u = K_E + \epsilon_{2t}\) \(\quad\)
  \emph{(terminal scrap value, absorbing)}
\end{itemize}
\end{frame}

\begin{frame}{4-Parameter Baseline: Transition \& Likelihood}
\phantomsection\label{parameter-baseline-transition-likelihood}
\BerkeleyLightMode

\textbf{Transitions:} Continue \(\Rightarrow a_t+1\); \(\;\) Retrofit
\(\Rightarrow a=0\); \(\;\) Exit \(\Rightarrow\) absorbing.

\[P(d_t \mid a_t ; \theta) = \frac{\exp(v(a_t, d_t))}{\sum_{j=0}^{2} \exp(v(a_t, j))}\]

Conditional independence of \(\epsilon_t\) yields closed-form logistic
choice probabilities. MLE over observed panel of facility actions.
\end{frame}

\begin{frame}{8-Parameter Heterogeneity: Utility Specification}
\phantomsection\label{parameter-heterogeneity-utility-specification}
\BerkeleyLightMode

\(\theta_{het} = (\kappa_{\mathrm{SW}}, \kappa_{\mathrm{DW}}, K_{\mathrm{SW}}, K_{\mathrm{DW}}, \gamma_{\mathrm{price,FF}}, \gamma_{\mathrm{price,RB}}, \gamma_{\mathrm{risk,FF}}, \gamma_{\mathrm{risk,RB}})\)

\begin{itemize}
